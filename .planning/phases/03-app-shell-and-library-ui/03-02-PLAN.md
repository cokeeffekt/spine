**Alpine.js store registration** — in a `<script>` tag BEFORE the Alpine CDN script. Register all three stores inside `alpine:init` listener (per Research Pattern 1):

```javascript
document.addEventListener('alpine:init', () => {
  Alpine.store('auth', {
    loggedIn: false,
    username: '',
    role: ''
  })

  Alpine.store('app', {
    view: 'login'
  })

  Alpine.store('library', {
    books: [],
    query: '',
    selectedBook: null,
    loading: false,

    async loadBooks() {
      this.loading = true
      try {
        const res = await fetch('/api/books')
        if (res.status === 401) {
          Alpine.store('auth').loggedIn = false
          Alpine.store('app').view = 'login'
          return
        }
        this.books = await res.json()
      } catch (e) {
        console.error('Failed to load books:', e)
      } finally {
        this.loading = false
      }
    },

    async selectBook(id) {
      try {
        const res = await fetch('/api/books/' + id)
        if (res.status === 401) {
          Alpine.store('auth').loggedIn = false
          Alpine.store('app').view = 'login'
          return
        }
        this.selectedBook = await res.json()
        Alpine.store('app').view = 'detail'
      } catch (e) {
        console.error('Failed to load book:', e)
      }
    },

    get filteredBooks() {
      if (!this.query) return this.books
      const q = this.query.toLowerCase()
      return this.books.filter(b =>
        b.title.toLowerCase().includes(q) ||
        (b.author && b.author.toLowerCase().includes(q))
      )
    }
  })
})

function formatDuration(sec) {
  const h = Math.floor(sec / 3600)
  const m = Math.floor((sec % 3600) / 60)
  return h + 'h ' + m + 'm'
}

function formatChapterDuration(startSec, endSec) {
  const dur = endSec - startSec
  const m = Math.floor(dur / 60)
  const s = Math.floor(dur % 60)
  return m + ':' + (s < 10 ? '0' : '') + s
}
```
