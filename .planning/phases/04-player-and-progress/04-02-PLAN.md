**1. Add sleep timer methods to `$store.player` store**

Add these methods to the player store object:

```javascript
setSleepTimer(mode) {
  this._clearSleepTimer()
  this.sleepMode = mode
  if (typeof mode === 'number') {
    this.sleepTimerEndsAt = Date.now() + mode * 60 * 1000
    this.sleepTimer = setTimeout(() => this._sleepStop(), mode * 60 * 1000)
  }
  // 'chapter' mode is detected in _onTimeUpdate — no setTimeout needed
},

cancelSleepTimer() {
  this._clearSleepTimer()
  this.sleepMode = null
  this.sleepTimerEndsAt = null
},

_sleepStop() {
  window._spineAudio.pause()  // triggers _onPause which saves progress
  this._clearSleepTimer()
  this.sleepMode = null
  this.sleepTimerEndsAt = null
},

_clearSleepTimer() {
  if (this.sleepTimer) { clearTimeout(this.sleepTimer); this.sleepTimer = null }
},
```

Add `sleepTimerEndsAt: null` to the store's state properties.

**6. Add keyboard shortcut listener (per D-14)**

Add a `document.addEventListener('keydown', ...)` call AFTER the `alpine:init` listener block (but still inside the `<script>` tag, before the Alpine CDN script tag). The listener:

```javascript
document.addEventListener('keydown', (e) => {
  // Don't intercept when typing in form inputs (per research Pitfall 4)
  if (['INPUT', 'SELECT', 'TEXTAREA'].includes(e.target.tagName)) return
  const player = Alpine.store('player')
  if (!player.book) return

  switch (e.code) {
    case 'Space':
      e.preventDefault()  // prevent page scroll
      player.togglePlay()
      break
    case 'ArrowLeft':
      e.preventDefault()
      player.skip(-10)  // per D-14: arrow keys seek 10s
      break
    case 'ArrowRight':
      e.preventDefault()
      player.skip(10)
      break
  }
})
```

**7. Add Media Session handlers (per D-14: play, pause, nexttrack, previoustrack)**

Add after the keydown listener. Per D-14, media keys (MediaPlayPause, MediaTrackNext, MediaTrackPrevious) must work if available. Phase 5 covers lock-screen *display* (title, author, cover, scrubber via LOCK-01/02/03) but basic media key control belongs here.

```javascript
if ('mediaSession' in navigator) {
  navigator.mediaSession.setActionHandler('play', () => {
    const player = Alpine.store('player')
    if (player.book && !player.playing) player.togglePlay()
  })
  navigator.mediaSession.setActionHandler('pause', () => {
    const player = Alpine.store('player')
    if (player.book && player.playing) player.togglePlay()
  })
  navigator.mediaSession.setActionHandler('nexttrack', () => {
    const player = Alpine.store('player')
    if (player.book && player.currentChapterIdx < player.book.chapters.length - 1) {
      player.jumpToChapter(player.currentChapterIdx + 1)
    }
  })
  navigator.mediaSession.setActionHandler('previoustrack', () => {
    const player = Alpine.store('player')
    if (player.book && player.currentChapterIdx > 0) {
      player.jumpToChapter(player.currentChapterIdx - 1)
    }
  })
}
```
