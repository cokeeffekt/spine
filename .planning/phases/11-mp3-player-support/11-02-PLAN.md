    **Step 2: Modify `startDownload(book)` for MP3 books (per D-10, D-11).**

    Add format branching at the top of the try block. For MP3 books, fetch each track sequentially and cache individually:

    ```javascript
    async startDownload(book) {
      if (this.states[book.id]?.status === 'downloading') return
      this.states[book.id] = { status: 'downloading', progress: 0, sizeBytes: 0 }

      const controller = new AbortController()
      this._controllers[book.id] = controller

      try {
        if (book.format === 'mp3') {
          // MP3: download each track individually (D-10, D-11)
          const totalTracks = book.chapters.length
          let downloaded = 0
          let totalBytes = 0
          const cache = await caches.open('spine-audio')

          for (let i = 0; i < totalTracks; i++) {
            const url = trackUrl(book.id, i)
            const res = await fetch(url, { signal: controller.signal })
            if (!res.ok) throw new Error('HTTP ' + res.status + ' for track ' + i)
            const blob = await res.blob()
            const fullResponse = new Response(blob, {
              status: 200,
              headers: {
                'Content-Type': 'audio/mpeg',
                'Content-Length': String(blob.size),
                'Accept-Ranges': 'bytes',
              }
            })
            await cache.put(url, fullResponse)
            downloaded++
            totalBytes += blob.size
            // D-11: progress = tracks downloaded / total tracks
            this.states[book.id] = {
              status: 'downloading',
              progress: downloaded / totalTracks,
              sizeBytes: totalBytes
            }
          }

          // Save with format and trackCount for cleanup and init reconciliation (Pitfall 4, 5)
          await downloadDB.save(book.id, { sizeBytes: totalBytes, downloadedAt: Date.now(), format: 'mp3', trackCount: totalTracks })
          this.states[book.id] = { status: 'complete', progress: 1, sizeBytes: totalBytes }
        } else {
          // m4b: existing single-URL download (unchanged)
          const res = await fetch('/api/books/' + book.id + '/audio', { signal: controller.signal })
          if (!res.ok) throw new Error('HTTP ' + res.status)

          const totalBytesHeader = parseInt(res.headers.get('Content-Length') || '0', 10)
          const reader = res.body.getReader()
          const chunks = []
          let loaded = 0

          while (true) {
            const { done, value } = await reader.read()
            if (done) break
            chunks.push(value)
            loaded += value.byteLength
            this.states[book.id] = {
              status: 'downloading',
              progress: totalBytesHeader > 0 ? loaded / totalBytesHeader : 0,
              sizeBytes: loaded
            }
          }

          const blob = new Blob(chunks, { type: 'audio/mp4' })
          const fullResponse = new Response(blob, {
            status: 200,
            headers: {
              'Content-Type': 'audio/mp4',
              'Content-Length': String(loaded),
              'Accept-Ranges': 'bytes',
            }
          })
    ```
