    **Part A: Add $store.downloads Alpine store** (in the alpine:init listener block in index.html, after $store.player):

    ```javascript
    Alpine.store('downloads', {
      // Reactive state: { [bookId]: { status: 'idle'|'downloading'|'complete'|'error', progress: 0-1, sizeBytes: 0 } }
      states: {},
      _controllers: {},  // AbortControllers, non-reactive (stored directly on this)

      async init() {
        // Restore from IndexedDB on page load
        const keys = await downloadDB.getAllKeys()
        const cache = await caches.open('spine-audio')
        for (const bookId of keys) {
          const entry = await cache.match('/api/books/' + bookId + '/audio')
          if (entry) {
            const meta = await downloadDB.get(bookId)
            this.states[bookId] = { status: 'complete', progress: 1, sizeBytes: meta ? meta.sizeBytes : 0 }
          } else {
            // Stale IndexedDB entry — cache was evicted
            await downloadDB.delete(bookId)
          }
        }
        // Request persistent storage silently on first load if any downloads exist
        if (keys.length > 0 && navigator.storage && navigator.storage.persist) {
          navigator.storage.persist().catch(() => {})
        }
      },

      // D-14: Proactively cache ALL cover art when library loads.
      // Called fire-and-forget from $store.library.load() after books array arrives.
      // Uses Cache Storage 'spine-covers' matching the service worker's CacheFirst route.
      async cacheAllCovers(books) {
        try {
          const coverCache = await caches.open('spine-covers')
          const urls = books.map(b => '/api/books/' + b.id + '/cover')
          // Filter out already-cached URLs to avoid redundant fetches
          const uncached = []
          for (const url of urls) {
            const existing = await coverCache.match(url)
            if (!existing) uncached.push(url)
          }
          if (uncached.length > 0) {
            await coverCache.addAll(uncached)
          }
        } catch (err) {
          // Fire-and-forget: log but don't block library loading
          console.warn('Cover pre-cache failed (non-fatal):', err.message)
        }
      },

      isDownloaded(bookId) { return this.states[bookId]?.status === 'complete' },
      isDownloading(bookId) { return this.states[bookId]?.status === 'downloading' },
      getProgress(bookId) { return this.states[bookId]?.progress ?? 0 },
      getSizeBytes(bookId) { return this.states[bookId]?.sizeBytes ?? 0 },

      async startDownload(book) {
        if (this.states[book.id]?.status === 'downloading') return
        this.states[book.id] = { status: 'downloading', progress: 0, sizeBytes: 0 }

        const controller = new AbortController()
        this._controllers[book.id] = controller

        try {
          const res = await fetch('/api/books/' + book.id + '/audio', { signal: controller.signal })
          if (!res.ok) throw new Error('HTTP ' + res.status)

          const totalBytes = parseInt(res.headers.get('Content-Length') || '0', 10)
          const reader = res.body.getReader()
          // Keep chunks in closure, NOT on reactive store (Pitfall 6 from research)
          const chunks = []
          let loaded = 0

          while (true) {
            const { done, value } = await reader.read()
            if (done) break
            chunks.push(value)
            loaded += value.byteLength
            this.states[book.id] = {
              status: 'downloading',
              progress: totalBytes > 0 ? loaded / totalBytes : 0,
              sizeBytes: loaded
            }
          }

          // Reassemble and store in Cache Storage
          const blob = new Blob(chunks, { type: 'audio/mp4' })
          const fullResponse = new Response(blob, {
            status: 200,
            headers: {
              'Content-Type': 'audio/mp4',
              'Content-Length': String(loaded),
              'Accept-Ranges': 'bytes',
            }
          })
          const cache = await caches.open('spine-audio')
          await cache.put('/api/books/' + book.id + '/audio', fullResponse)

          await downloadDB.save(book.id, { sizeBytes: loaded, downloadedAt: Date.now() })
          this.states[book.id] = { status: 'complete', progress: 1, sizeBytes: loaded }

          // Request persistent storage on first successful download
          if (navigator.storage && navigator.storage.persist) {
            navigator.storage.persist().catch(() => {})
          }
        } catch (err) {
          if (err.name === 'AbortError') {
            // Cancelled by user — cleanup done in cancelDownload
          } else {
            console.error('Download failed:', err)
            this.states[book.id] = { status: 'error', progress: 0, sizeBytes: 0 }
            await this._cleanup(book.id)
          }
        } finally {
          delete this._controllers[book.id]
        }
      },

      async cancelDownload(bookId) {
        const controller = this._controllers[bookId]
        if (controller) controller.abort()
        await this._cleanup(bookId)
        this.states[bookId] = { status: 'idle', progress: 0, sizeBytes: 0 }
      },

      async deleteDownload(book) {
        // D-10: native confirm() dialog
        const size = formatBytes(this.states[book.id]?.sizeBytes || 0)
        if (!confirm('Delete download for ' + book.title + '? (' + size + ')')) return
        await this._cleanup(book.id)
        this.states[book.id] = { status: 'idle', progress: 0, sizeBytes: 0 }
      },

      async _cleanup(bookId) {
        const cache = await caches.open('spine-audio')
        await cache.delete('/api/books/' + bookId + '/audio')
        await downloadDB.delete(bookId)
      }
    })
    ```
